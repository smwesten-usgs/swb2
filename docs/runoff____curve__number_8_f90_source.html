175! elemental function update_curve_number_fn( iLanduseIndex, iSoilsIndex, fSoilStorage, &
176! fSoilStorage_Max, fCFGI ) result( CN_adj )
177
178! integer (c_int), intent(in) :: iLanduseIndex
179! integer (c_int), intent(in) :: iSoilsIndex
180! real (c_float), intent(in) :: fSoilStorage
181! real (c_float), intent(in) :: fSoilStorage_Max
182! real (c_float), intent(in) :: fCFGI
183! real (c_float) :: CN_adj
184
185! ! [ LOCALS ]
186! real (c_float) :: Pf
187! real (c_float) :: CN_base
188! real (c_float) :: fFraction_FC ! fraction of field capacity
189! real (c_float) :: frac
190
191! fFraction_FC = 0.0
192! if (fSoilStorage_Max > 0.0) fFraction_FC = min( 1.0_c_float, fSoilStorage / fSoilStorage_Max )
193
194! ! Correct the curve number...
195! !
196! ! current thinking on this: if the soil moisture is at average levels or *greater*
197! ! *and* the CFGI indicates frozen ground conditions, assume that runoff enhancement
198! ! is probable. If the soils froze under relatively dry conditions, assume that some pore
199! ! space is open in the frozen ground.
200! if( fCFGI > CFGI_LL .and. fFraction_FC > 0.5 ) then
201
202! Pf = prob_runoff_enhancement( fCFGI )
203
204! ! use probability of runoff enhancement to calculate a weighted
205! ! average of curve number under Type II vs Type III antecedent
206! ! runoff conditions
207! CN_adj = CN_ARCII(iLanduseIndex, iSoilsIndex) * ( 1.0_c_float - Pf ) &
208! + CN_ARCIII(iLanduseIndex, iSoilsIndex) * Pf
209
210! elseif ( fFraction_FC > 0.5_c_float ) then ! adjust upward toward AMC III
211
212! ! we want a number ranging from 0 to 1 indicating the relative split
213! ! between CN AMCII and CN AMCIII
214! frac = ( fFraction_FC - 0.5_c_float ) / 0.5_c_float
215
216! CN_adj = CN_ARCII(iLanduseIndex, iSoilsIndex) * ( 1.0_c_float - frac ) &
217! + CN_ARCIII(iLanduseIndex, iSoilsIndex) * frac
218
219! elseif ( fFraction_FC < 0.5_c_float ) then ! adjust downward toward AMC I
220
221! ! we want a number ranging from 0 to 1 indicating the relative split
222! ! between CN AMCII and CN AMCI
223! !
224! ! ex. fFraction_FC = 0.48; frac = 2.0 * 0.48 = 0.96
225! ! CN_adj = 0.96 ( CN_ARCII ) + ( 1.0 - 0.96 ) * CN_ARCIII
226! !
227! frac = fFraction_FC / 0.5_c_float
228
229! CN_adj = CN_ARCI(iLanduseIndex, iSoilsIndex) * ( 1.0_c_float - frac ) &
230! + CN_ARCII(iLanduseIndex, iSoilsIndex) * frac
231
232! else
233
234! CN_adj = CN_ARCII(iLanduseIndex, iSoilsIndex)
235
236! endif
237
238
239! ! ensure that whatever modification have been made to the curve number
240! ! remain within reasonable bounds
241! CN_adj = MIN( CN_adj, 100.0_c_float )
242! CN_adj = MAX( CN_adj, 0.0_c_float )
243
244! end function update_curve_number_fn

260 ! [ LOCALS ]

262 real (c_float) :: frac

267 associate( cn_i => cn_arci( ilanduseindex, isoilsindex ), &
268 cn_ii => cn_arcii( ilanduseindex, isoilsindex ), &
269 cn_iii => cn_arciii( ilanduseindex, isoilsindex ) )
270
271 ! Correct the curve number...
272
273 if( ( fcfgi > cfgi_ll ) .and. ( fsoilstorage_max > 0.0_c_float ) ) then
274
275 pf = prob_runoff_enhancement( fcfgi )
276
277 ! use probability of runoff enhancement to calculate a weighted
278 ! average of curve number under Type II vs Type III antecedent
279 ! runoff conditions
280 cn_adj = cn_ii * (1-pf) + cn_iii * pf
281
282 else if ( it_is_growing_season ) then
283
284 if ( finflow < amc_dry_growing ) then
285
286 cn_adj = cn_i
287
288 else if ( finflow >= amc_dry_growing &
289 .and. finflow < amc_wet_growing ) then
290
291 cn_adj = cn_ii
292
293 else
294
295 cn_adj = cn_iii
296
297 end if
298
299 else ! dormant (non-growing) season
300
301 if ( finflow < amc_dry_dormant ) then
302
303 cn_adj = cn_i
304
305 else if ( finflow >= amc_dry_dormant &
306 .and. finflow < amc_wet_dormant ) then
307
308 cn_adj = cn_ii
309
310 else
311
312 cn_adj = cn_iii
313
314 end if
315
316 end if
317
318 end associate
319
320 ! ensure that whatever modification have been made to the curve number
321 ! remain within reasonable bounds
322 cn_adj = min(cn_adj,100.0_c_float)
323 cn_adj = max(cn_adj,30.0_c_float)
324

396
397 ! [ LOCALS ]
398! real (c_float) :: CN_05
399 real (c_float) :: smax
400 real (c_float) :: cn_adj
401
402 curve_num_adj = update_curve_number_fn( landuse_index, soil_group, &
403 cell_index, &
404 it_is_growing_season, &
405 soil_storage_max, &
406 continuous_frozen_ground_index )
407
408 smax = ( 1000.0_c_float / curve_num_adj ) - 10.0_c_float
409
410! ! Equation 9, Hawkins and others, 2002
411! CN_05 = 100_c_float / &
412! ((1.879_c_float * ((100.0_c_float / CN_adj ) - 1.0_c_float )**1.15_c_float) + 1.0_c_float)
413
414 ! Equation 8, Hawkins and others, 2002
415 ! adjust Smax for alternate initial abstraction amount
416 smax = 1.33_c_float * ( smax**1.15_c_float )
417
418 ! now consider runoff if Ia ~ 0.05S
419 if ( inflow > 0.05_c_float * smax ) then
420 runoff = ( inflow - 0.05_c_float * smax )**2 / ( inflow + 0.95_c_float * smax )
421 else
422 runoff = 0.0_c_float
423 end if
