28 !> Adjust the depletion fraction based on current reference ET0.
29 !!
30 !! From FAO-56: "The fraction p is a function of the evaporation power of the atmosphere.
31 !! At low rates of ETc, the p values listed in Table 22 are higher than at high rates of ETc.
32 !! For hot dry weather conditions, where ETc is high, p is 10-25% less than the values
33 !! presented in Table 22, and the soil is still relatively wet when the stress starts to occur.
34 !! When the crop evapotranspiration is low, p will be up to 20% more than the listed values.
35 !!
36 !! @param[in] p_table_22 This is the unadjusted depletion fraction value; FAO-56
37 !! table 22 gives values of the depletion fraction relative to a reference ET0 value of 5mm.
38 !! @param[in] reference_et0 The reference ET0 to which the depletion fraction will be
39 !! adjusted.
40 !! @note Discussed as a footnote to Table 22, FAO-56, Allen and others.
41 !! See @ref http://www.fao.org/docrep/x0490e/x0490e0e.htm#TopOfPage for details.
42
43 elemental function adjust_depletion_fraction_p( landuse_index, reference_et0 ) result( p )
44
45 integer ( c_int), intent(in) :: landuse_index
46 real (c_double), intent(in) :: reference_et0
47 real (c_double) :: p
48
49 p = real(depletion_fraction( landuse_index ),c_double) + 0.04_c_double &
50 * ( 5.0_c_double - in_to_mm( reference_et0 ) )
51
52 p = min( p, 0.8_c_double )
53 p = max( p, 0.1_c_double )
54
55 end function adjust_depletion_fraction_p

164!> This function estimates the fraction of the ground covered by
165!> vegetation during the growing season
166!> @note Implemented as equation 76, FAO-56, Allen and others
167impure elemental function calculate_fraction_exposed_and_wetted_soil_fc( landuse_index, Kcb, current_plant_height) result ( few )
168
169 ! [ ARGUMENTS ]
170 integer (c_int), intent(in) :: landuse_index
171 real (c_float), intent(in) :: kcb
172 real (c_float), intent(in) :: current_plant_height
173
174 ! [ RESULT ]
175 real (c_float) :: few
176
177 ! [ LOCALS ]
178 real (c_float) :: fc
179 real (c_double) :: numerator
180 real (c_double) :: denominator
181 real (c_double) :: exponent
182
183 numerator = max(kcb - kcb_l( kcb_min, landuse_index), 0.0_c_double)
184 denominator = kcb_l( kcb_mid, landuse_index) - kcb_l( kcb_min, landuse_index)
185 exponent = 1.0_c_double + 0.5_c_double * current_plant_height * m_per_foot
186
187 if( denominator > 0.0_c_double ) then
188 fc = ( numerator / denominator ) ** exponent
189 else
190 fc = 1.0_c_float
191 endif
192
193 fc = max(fc, min_fraction_covered_soil( landuse_index ))
194 few = clip(1.0_c_float - fc, minval=0.05, maxval=1.0)
195
196end function calculate_fraction_exposed_and_wetted_soil_fc

377 adjusted_depletion_fraction_p, &

403 ! actual_et needs to respect limitations on the amount of remaining soil moisture present
