396 impure elemental function update_crop_coefficient_date_as_threshold( iLanduseIndex ) &
397 result(kcb)
398
399 integer (c_int), intent(in) :: ilanduseindex
400 real (c_float) :: kcb
401
402 ! [ LOCALS ]
403 real (c_double) :: ffrac
404
405 if ( kcb_method( ilanduseindex ) == kcb_method_monthly_values ) then
406
407 kcb = kcb_l( sim_dt%curr%iMonth, ilanduseindex )
408
409 else
410
411 ! define shorthand variable names for remainder of function
412 associate( date_ini => growth_stage_date( enddate_ini, ilanduseindex ), &
413 date_dev => growth_stage_date( enddate_dev, ilanduseindex ), &
414 date_mid => growth_stage_date( enddate_mid, ilanduseindex ), &
415 date_late => growth_stage_date( enddate_late, ilanduseindex ), &
416 date_fallow => growth_stage_date( enddate_fallow, ilanduseindex ), &
417 kcb_ini => kcb_l(kcb_ini, ilanduseindex), &
418 kcb_mid => kcb_l(kcb_mid, ilanduseindex), &
419 kcb_min => kcb_l(kcb_min, ilanduseindex), &
420 plantingdate => growth_stage_date( planting_date, ilanduseindex), &
421 kcb_end => kcb_l(kcb_end, ilanduseindex), &
422 current_date => sim_dt%curr )
423
424 ! now calculate Kcb for the given landuse
425
426 if( current_date > date_late ) then
427
428 kcb = kcb_min
429
430 elseif ( current_date > date_mid ) then
431
432 ffrac = ( current_date - date_mid ) / ( date_late - date_mid )
433
434 kcb = kcb_mid * (1.0_c_double - ffrac) + kcb_end * ffrac
435
436 elseif ( current_date > date_dev ) then
437
438 kcb = kcb_mid
439
440 elseif ( current_date > date_ini ) then
441
442 ffrac = ( current_date - date_ini ) / ( date_dev - date_ini )
443
444 kcb = kcb_ini * (1.0_c_double - ffrac) + kcb_mid * ffrac
445
446 elseif ( current_date >= plantingdate ) then
447
448 kcb = kcb_ini
449
450 else
451
452 kcb = kcb_min
453
454 endif
455
456 end associate
457
458 end if
459
460end function update_crop_coefficient_date_as_threshold

464pure elemental function crop_coefficients_fao56_calculate_kcb_max(wind_speed_meters_per_sec, &
465 relative_humidity_min_pct, &
466 Kcb, &
467 plant_height_meters) result(kcb_max)
468
469 real (c_float), intent(in) :: wind_speed_meters_per_sec
470 real (c_float), intent(in) :: relative_humidity_min_pct
471 real (c_float), intent(in) :: kcb
472 real (c_float), intent(in) :: plant_height_meters
473
474 real (c_float) :: kcb_max
475 real (c_double) :: u2
476 real (c_double) :: rhmin
477 real (c_double) :: plant_height
478
479 ! Limits are as suggested on page 123 of FAO-56 with respect to
480 ! modifying mid-season KCB_mid values
481 rhmin = clip( relative_humidity_min_pct, minval=20., maxval=80. )
482 u2 = clip(wind_speed_meters_per_sec, minval=1., maxval=6.)
483 plant_height = clip(plant_height_meters, minval=1., maxval=10.)
484
485 ! equation 72, FAO-56, p 199
486 kcb_max = max( 1.2_c_double + ( (0.04_c_double * (u2 - 2._c_double) &
487 - 0.004_c_double * (rhmin - 45._c_double) ) ) &
488 * (plant_height_meters/3._c_double)**0.3_c_double, &
489 kcb + 0.05_c_double )

512 associate( gdd_ini_l => growth_stage_gdd( gdd_ini, ilanduseindex ), &
513 gdd_dev_l => growth_stage_gdd( gdd_dev, ilanduseindex ), &
514 gdd_mid_l => growth_stage_gdd( gdd_mid, ilanduseindex ), &

523 current_doy => sim_dt%curr%getDayOfYear() )
524
525 ! now calculate Kcb for the given landuse
526 if( fgdd > gdd_late_l ) then
527
528 fkcb = kcb_min
529
530 elseif ( fgdd > gdd_mid_l ) then
531
532 ffrac = ( fgdd - gdd_mid_l ) / ( gdd_late_l - gdd_mid_l )
533
534 fkcb = kcb_mid * (1.0_c_double - ffrac) + kcb_end * ffrac
535
536 elseif ( fgdd > gdd_dev_l ) then
537
538 fkcb = kcb_mid
539
540 elseif ( fgdd > gdd_ini_l ) then
541
542 ffrac = ( fgdd - gdd_ini_l ) / ( gdd_dev_l - gdd_ini_l )
543
544 fkcb = kcb_ini * (1_c_double - ffrac) + kcb_mid * ffrac
545
546 elseif ( (plantingdoy > 0) .and. (current_doy >= plantingdoy) ) then
547
548 ! if there is a valid value for the Planting Date, use it
549 fkcb = kcb_ini
550
551 elseif ( (gdd_plant_l >= 0.) .and. (fgdd >= gdd_plant_l) ) then
552
553 fkcb = kcb_ini
554
555 else
556
557 fkcb = kcb_min
